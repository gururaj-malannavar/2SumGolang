import "fmt"

func twoSum(nums []int, target int) []int {
    
    ans := make([]int, 2)
    
    map1 := make(map[int]int)
    
    
    for key, val := range nums {
        
        if pos, ok := map1[target-val]; ok {
            ans[0] = key
            ans[1] = pos
            return ans
        }
        map1[val] = key
        
    }
    return ans
}
